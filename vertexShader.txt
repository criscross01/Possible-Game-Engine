#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in vec3 aNormal;

uniform mat4 transform;
uniform mat4 view;
uniform mat4 persp;

out vec3 Normal;
out vec3 FragPos;

void main()
{
    FragPos = vec3(transform * vec4(aPos, 1.0));
    Normal = aNormal;

    gl_Position = persp * view * transform * vec4(aPos, 1.0f);
}
